# leetcode 78 子集

```C++
    vector<vector<int>> subsets(vector<int>& nums) {
        //backtracking(nums, 0);
        vector<vector<int>> res;
        int state = 1 << nums.size();
        //[0, state)
        for(int i = 0; i < state; i++){
            vector<int> tmp;
            for(int j = 0; j < nums.size(); j++){
                int b = (i >> j) & 1;
                if(b){
                    tmp.push_back(nums[j]);
                }
            }
            res.push_back(tmp);
        }
        return res;
    }
};
```

# leetcode 191 位1的个数

```C++
class Solution {
public:
    int hammingWeight(uint32_t n) {
        int res = 0;
        while(n){
            int lowbit = n&(-n);
            res ++;
            n -= lowbit;
        }
        return res;
    }
};
```

# leetcode 1442 形成两个异或相等数组的三元组数目

```C++
class Solution {
public:
    int countTriplets(vector<int>& arr) {
        int res = 0;
        vector<int> s(arr.size());
        s[0] = arr[0];
        //类似于前缀和
        for(int i = 1; i < arr.size(); i++){
            s[i] = s[i - 1] ^ arr[i];
        }

        for(int i = 0; i < arr.size(); i++){
            for(int k = i + 1; k < arr.size(); k++){
                //[i,k]的抑或结果
                int sum = i == 0 ? s[k] : s[k] ^ s[i - 1];
                if(sum  == 0){
                    res += (k - i);
                }
            }
        }
        return res;
    }
};
```

# leetcode 476 数字的补数

```C++
class Solution {
public:
    int findComplement(int n) {
        int res = 0;
        int tmp = n;
        for(int i = 0; i < 31; i++){
            int b = (n >> i) & 1;
            //判断是否到前导零
            if(b)
                tmp -= 1 << i;
            if(!tmp)
                break;
            if(!b)
                res |= (1 << i);
        }
        return res;
    }
};
//11010的补数为
//00101相加则为
//11111，因此只需要找到其最后一位1，因此可以使用lowbit
class Solution {
public:
    int findComplement(int n) {
        int tmp = n;
        while(n){
            int lowbit = n & (-n);
            n -= lowbit;
            if(n == 0){
                return 2LL * lowbit - 1 - tmp;
            }
        }
        return 0;
    }
};
```

# leetcode 201 数字范围按位与

```C++
class Solution {
public:
    int rangeBitwiseAnd(int l, int r) {
        int res = 0;
        //从高位开始，只有当最大和最小的高位为1时才需要计算结果，如果一个0一个1则结果全是0
        for(int i= 30; i >= 0; i--){
            int bl = (l >> i) & 1;
            int br = (r >> i) & 1;
            if(bl && br)
                res |= 1 << i;
            else if(!bl && br)break;
        }
        return res;
    }
};
```

# leetcode 137 只出现一次的数字II

```C++
class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int res = 0;
        //按位处理
        for(int i = 0; i < 32; i++){
            //统计这一位里0、1的数量，但是不用关心0的数量
            int cnt = 0;
            for(auto item : nums){
                int b = (item >> i) & 1;
                //if(b) cnt++; 等价于下面的
                cnt += b;
            }
            if(cnt % 3)
                res |= (1 << i);
        }
        return res;
    }
};
```
