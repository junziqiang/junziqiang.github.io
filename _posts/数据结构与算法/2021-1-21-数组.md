## 数组的存放方式

一维数组的内存地址是连续的，二维数组是以行优先存储

```C++
#include<bits/stdc++.h>
using namespace std;

int main(){
        
    int A[3];
    for(int i = 0; i < 3; ++i){
        cout<<"i="<<i<<"时地址" << &A[i]<< " " ;
    }
    cout<<endl;
    int B[3][4];
    for(int i = 0; i < 3; i++){
        for(int j = 0; j < 4; j++){
            cout<<"i=" <<i<<"地址 "<<&B[i][j]<<" ";
        }
        cout<<endl;
    }
    return 0;
}
```

## 数组二分

### leetcode35 搜索插入的位置

```C++
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int n = nums.size();
        int left = 0;
        int right = n;

        while(left < right){
            int middle = left + (right - left) / 2;
            if(nums[middle] < target){
                left = middle + 1;
            }else if(nums[middle] > target){
                right = middle;
            }else if(nums[middle] == target){
                return middle;
            }
        }
        //1.要插入的情况，在所有数据的前面[0,0)
        //2.在所有数据的后面[n,n)
        //3.在中间
        //若是采用前闭后闭区间进行写二分此时返回的结果应该是right + 1；
        return right;
    }
};
```

**在写二分的时候应该坚持循环不变量，也就是区间的定义**

## leetcode27 移除元素

> 快慢指针通过一个快指针和慢指针在一个for循环中完成两个for循环的工作

![数组移除元素](/img/数据结构与算法/数组移除元素.gif)

```C++
//快慢指针解法
class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int slowIndex = 0;
        for(int fastIndex = 0; fastIndex < nums.size(); fastIndex++){
            if(nums[fastIndex] != val){
                nums[slowIndex]=nums[fastIndex];
                slowIndex++;
            }
        }
        return slowIndex;
    }
};
```

## leetcode209 长度最小的子数组

> 根据滑动窗口的框架

```C++
class Solution {
public:
    int minSubArrayLen(int s, vector<int>& nums) {
        int left = 0, right = 0;
        int sum = 0;
        int res = INT_MAX;
        int subLength = 0;
        while(right < nums.size()){
            //移入窗口
            sum+=nums[right];
            //右移窗口
            right++;
            //进行窗口中的一系列更新
            //无
            //判断是否缩小窗口
            while(sum >= s){
                //移出最左侧的数据
                sum -=nums [left];
                //缩小窗口
                left++;
                subLength = right - left + 1;
                res = res < subLength ? res:subLength;

            }
        }
        return res == INT_MAX ? 0 : res;
    }
};
```
