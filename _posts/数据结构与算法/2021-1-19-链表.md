# 链表的反转

## leetcode206反转链表

```C++
class Solution {
public:
    ListNode* reverseList(ListNode* head) {
       if(head == nullptr || head->next == nullptr)
            return head;
        ListNode* last = reverseList(head->next);
        head->next->next = head;
        head->next = nullptr;
        return last; 
    }
};

//反转前n个节点
ListNode* sucessor = nullptr;
ListNode* reverseList(ListNode* head, int n){
    if(n == 1){
        //记录n+1的节点，接就是后继节点，记录的方法则是使用一个全局变量，记录前驱节点的方法也是一样
        sucessor = head->next;
        return head;
    }
    ListNode* last = reverseList(head->next, n-1);
    head->next->next = head;
    head->next = sucessor;
    return last;
}
```

## leetcode92反转一部分链表

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
ListNode* sucessor = nullptr;
ListNode* reverseList(ListNode* head, int n){
    if(n == 1){
        //记录n+1的节点，接就是后继节点，记录的方法则是使用一个全局变量，记录前驱节点的方法也是一样
        sucessor = head->next;
        return head;
    }
    ListNode* last = reverseList(head->next, n-1);
    head->next->next = head;
    head->next = sucessor;
    return last;
}
    ListNode* reverseBetween(ListNode* head, int m, int n) {
        if(m == 1){
            return reverseList(head, n);
        }
        head->next = reverseBetween(head->next,m-1,n-1);
        return head;
    }
};

## leetcode25 k个一组反转链表

```C++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        if(head == nullptr)
            return head;
        //反转区间
        ListNode* start, *end;
        start = end = head;
        
        for(int i = 0; i < k; ++i){
            //不足k个不需要反转
            if(end == nullptr)
                return head;//注意返回head
            end = end->next;
        }
        //反转start到end之间的节点
        ListNode* newHead = reverse(start, end);
		//递归反转end后的k个节点，返回头节点，该头节点放在上一个start节点的next；
        start->next = reverseKGroup(end, k);
        return newHead;
    }
    //迭代进行反转链表
    //若反转[head,end)之间的节点，只需要加一个参数，while的终止条件修改一下即可
    ListNode* reverse(ListNode* head, ListNode* end){
        ListNode* pre = nullptr;
        ListNode* cur = head;
        ListNode* next = head;
        while(cur != end){
            next = cur->next;
            cur->next = pre;
            pre = cur;
            cur = next;
        }
        return pre;
    }


};
```
