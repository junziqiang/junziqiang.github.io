> leetcode20 有效的括号

```c++
#include<bits/stdc++.h>
using namespace std;

class Solution{
public:
    unordered_map<char,char> num = {
        {'(',')'},
        {'[',']'},
        {'{','}'}
    };
    bool isValid(string s){
        if(s.empty())
            return true;
        else if(s.size()%2)
            return false;
        else{
            stack<char> ss;
            for(int i = 0; i < s.size(); ++i){
                if(ss.empty())
                    ss.push(s[i]);
                else{
                    if(s[i] == num[ss.top()]){
                        ss.pop();
                    }else{
                        ss.push(s[i]);
                    }
                }
            }
            return ss.empty()
        }
        
    }
};
```

> leetcode856括号的分数

```c++
class Solution {
public:
    int scoreOfParentheses(string S) {
        stack<int> ss;
        for(char c : S){
            //左括号直接入栈
            if(c == '('){
                ss.push(0);//使用0代表左括号
            }
            //右括号的时候
            else{
                //栈顶是左括号的时候，直接将得分入栈
                if(ss.top() == 0){
                    ss.pop();
                    ss.push(1);
                }
                //栈顶不是左括号的时候，那就是(AB时，此时将所有得分相加，最后乘以2入栈
                else{
                    int score = 0;
                    while(ss.top() != 0){
                        score +=ss.top();
                        ss.pop();
                    }
                    ss.pop();
                    ss.push(score*2);
                }
            }
        }
        //最后的结果栈内都是得分，直接相加
        int score = 0;
        while(!ss.empty()){
            score+=ss.top();
            ss.pop();
        }
        return score;
    }
};
```

> leetcode150逆波兰式求值

```c++
class Solution {
public:
    int evalRPN(vector<string>& tokens) {
        stack<int> ss;
        for(auto c : tokens){
            if(c=="+" || c=="-"||c=="*"||c=="/"){
                int num1 = ss.top();
                ss.pop();
                int num2 = ss.top();
                ss.pop();
                if(c == "+")ss.push(num1+num2);
                if(c == "-")ss.push(num2-num1);
                if(c == "*")ss.push(num1*num2);
                if(c == "/")ss.push(num2/num1);
            }else{
                ss.push(atoi(c.c_str()));
            }
        }
        return ss.top();
    }
};
```

## 单调栈

- 单调栈分为单调递减栈和单调递减栈
  
  - 单调递增栈即栈内元素保持单调递增，同理单调递减栈时栈内元素保持递减
  
- 操作规则（单调递增栈为例）
  
  - 如果新的元素比栈顶元素大就入栈。如果新元素比较小那么就一直将栈内元素弹出到栈顶元素比新元素小为止。
  
- 加入这样的规则之后会有栈内元素都是递增的，当元素出栈时，说明这个新元素是出栈元素向后找第一个比其小的元素。

```c++
  stack<int> st;
  for(int i = 0; i < nums.size(); i++)
  {
  	while(!st.empty() && st.top() > nums[i])
  	{
  		st.pop();
  	}
  	st.push(nums[i]);
  }
```
> leetcode84 柱状图中最大的矩形

<b>单调递增栈</b>
```c++
/**
 * 给定 n 个非负整数，用来表示柱状图中各个柱子的高度。每个柱子彼此相邻，且宽度为 1 。

求在该柱状图中，能够勾勒出来的矩形的最大面积。
 * **/
#include<bits/stdc++.h>
using namespace std;

class Solution {
public:
    int largestRectangleArea(vector<int>& heights) {
        int ans = 0;
        stack<int> st;
        //加入边界判断
        heights.insert(heights.begin(),0);
        heights.push_back(0);
        for(int i = 0; i < heights.size(); ++i){
            //栈中存储下标，因为计算面积需要下标
            //当新的元素比较小的时候，此时计算面积
            while(!st.empty() && heights[st.top()] > heights[i]){
                //此时需要对每一个高度都求一次面积
                int curIndex = st.top();
                st.pop();
                //width也等于
                //left = st.top()+1;
                //right = i - 1;
                //width = right - left + 1;
                int width = i - st.top() - 1;
                int high = heights[curIndex];
                ans = max(ans,width*high);
            }
            st.push(i);
        }
        return ans;
    }
};
```

> leetcode42接雨水

```c++
class Solution {
public:
    int trap(vector<int>& height) {
        int ans = 0;
        stack<int> st;
        for(int i = 0; i < height.size(); ++i){
            while(!st.empty() && height[st.top()] < height[i]){
                int cur = st.top();
                st.pop();
                if(st.empty())
                    break;
                int r = i;
                int l = st.top();
                int h = min(height[r],height[l]) - height[cur];
                ans+=(r-l-1)*h;
            }
            st.push(i);
        }
        return ans;
    }
};
```
